## 4.졸업선물

### 완전탐색(브루트포스)이란?

완전탐색이란 모든 가능한 경우를 하나씩 시도하여 정확한 해를 찾아 문제를 해결하는 알고리즘

📌 느낀점
모든 상품이 할인 받았다고 가정하고 다 시도해봐야 했던 문제

### 나의 코드

```js
// 나의 코드
function solution(m, product) {
  let answer = 0;
  let n = product.length;
  product.sort((a, b) => a[0] + a[1] - (b[0] + b[1]));
  for (let i = 0; i < n; i++) {
    let money = m - (product[i][0] / 2 + product[i][1]); //할인된 금액을 뺀 나머지 금액
    let cnt = 1; //할인된거 삿으니 1
    for (let j = 0; j < n; j++) {
      if (j !== i && product[j][0] + product[j][1] > money) break;
      if (j !== i && product[j][0] + product[j][1] <= money) {
        money -= product[j][0] + product[j][1]; //배송비와 물건가격을 합친금액 빼기
        cnt++;
      }
    }
    answer = Math.max(answer, cnt);
  }
  return answer;
}

let arr = [
  [6, 6],
  [2, 2],
  [4, 3],
  [4, 5],
  [10, 3],
];
console.log(solution(28, arr));
```
