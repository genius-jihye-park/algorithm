function solution(m, arr) {
  let cnt = 0;
  for (let i = 0; i < arr.length - 1; i++) {
    let count = arr[i];
    for (let j = i + 1; j < arr.length; j++) {
      if (count + arr[j] > m) break;
      else if (count + arr[j] == m) {
        cnt++;
        break;
      } else count += arr[j];
    }
  }
  return cnt;
}

let a = [1, 2, 1, 3, 1, 1, 1, 2];
console.log(solution(6, a));

function solution(m, arr) {
  let answer = 0,
    lt = 0,
    sum = 0;
  for (let rt = 0; rt < arr.length; rt++) {
    sum += arr[rt];
    if (sum === m) answer++;
    while (sum >= m) {
      sum -= arr[lt++];
      if (sum === m) answer++;
    }
  }
  return answer;
}
