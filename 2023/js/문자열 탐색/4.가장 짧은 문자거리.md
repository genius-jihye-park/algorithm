function solution(s, t) {
  let result = [];
  let p = 0;
  for (let i = 0; i < s.length; i++) {
    if (s[i] == t) {
      p = 0;

      result.push(p);
    }
  }
  for (let i = s.length - 1; i >= 0; i--) {
    if (s[i] == t) {
      p = 0;
    } else {
      p++;
      result[i] = Math.min(result[i], p);
    }
  }
  return result;
}
