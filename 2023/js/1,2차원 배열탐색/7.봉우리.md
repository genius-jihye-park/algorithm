## 7.봉우리

### 1,2차원 배열 탐색

📌 느낀점
dx,dy 배열 먼저 만드는 것부터 시작하고 배열탐색하기.

### 나의 코드

```js
// 나의 코드
function solution(arr) {
  let answer = 0;
  let n = arr.length;
  let dx = [-1, 0, 1, 0]; //행:9시,6시,3시,12시 방향
  let dy = [0, 1, 0, -1]; //열

  for (let i = 0; i < n; i++) {
    for (let j = 0; j < n; j++) {
      let flag = 1;
      for (let k = 0; k < 4; k++) {
        let nx = i + dx[k]; //진행방향 행
        let ny = j + dy[k]; //진행방향 열
        if (
          nx >= 0 &&
          nx < n &&
          ny >= 0 &&
          ny < n &&
          arr[nx][ny] >= arr[i][j]
        ) {
          flag = 0;
          break;
        }
      }
      if (flag) answer++;
    }
  }
  return answer;
}

let arr = [
  [5, 3, 7, 2, 3],
  [3, 7, 1, 6, 1],
  [7, 2, 5, 3, 4],
  [4, 3, 6, 4, 1],
  [8, 7, 3, 5, 2],
];
console.log(solution(arr));
```
